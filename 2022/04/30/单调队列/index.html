单调队列

定义

维护一种序列, 可以从头和尾增加或删除元素(即双端队列), 且元素具有单调性.

实现

本质是维护具有单调性的最优决策集合, 该如何维护最优与单调的性质呢?

维护序列

该操作的本质是对原序列\(A\)扫描, 将当前的元素\(A_i\)与单调队列队尾比较, 如果\(A_i\)更优则将队尾换成\(A_i\).

//pseudo code
Queue q, Array a[];
int head = 0, tail = 0;
for each a[i] in a {
	while(队列过大) ++head;									//pop front
	//此时q[head]为当前最优
    while(head <= tail && q[tail]没有a[i]优) --tail;		//pop back
    q[++tail] = 当前值;
}

实现双端队列

std::deque

可以使用deque对序列进行操作.

using namespace std;

deque<int> q;

//入队
q.push_front(x);
q.push_back(x);

//出队
q.pop_front();
q.pop_back();

//取头尾
q.front();
q.back();

//遍历
for(int x: q) {
    cout << x << " ";
}
puts("");

数组模拟

可以使用两个指针\(head\), \(tail\)模拟.

需要注意的是数组中下标小的元素在队列前面, 同时指针\(head \le tail\).

例题

滑动窗口

给定一个长度为\(n\)的序列, 求其每个长度为\(k\)的连续子序列中的最大与最小值.

本题只需维护两个单调队列(一个递减, 队头为min; 另一个反之)即可.

需要注意的是要让窗口大小不超过\(k\), 这里采用在队列中存储数组下标的方法统计, 即保证\(head\)中存储的下标与当前扫描元素的下标之差小于\(k\).

#include <iostream>
#include <cstring>

using namespace std;

const int N = 1e6 + 100;

int q[N], a[N];
int n, k;

void getMin() {
	//维护递减序列, 队头即为答案
	int head = 0, tail = 0;
	for(int i = 1; i < k; ++i) {
		//Step 1: [1, k), push
		while(head <= tail && a[q[tail]] >= a[i]) --tail;
		q[++tail] = i;
	}
	for(int i = k; i <= n; ++i) {
		//Step 2: [k, n]
		//这个区间窗口可能超过k, 需要特判维护窗口大小
		while(head <= tail && a[q[tail]] >= a[i]) --tail;
		q[++tail] = i;
		while(q[head] <= i - k) ++head;
		cout << a[q[head]] << ' ';
	}
}

void getMax() {
	memset(q, 0, sizeof q);
	int head = 0, tail = 0;
	for(int i = 1; i < k; ++i) {
		while(head <= tail && a[q[tail]] <= a[i]) --tail;
		q[++tail] = i;
	}
	for(int i = k; i <= n; ++i) {
		while(head <= tail && a[q[tail]] <= a[i]) --tail;
		q[++tail] = i;
		if(q[head] <= i - k) ++head;
		cout << a[q[head]] << ' ';
	}
}

int main() {
	cin >> n >> k;
	for(int i = 1; i <= n; ++i) {
		cin >> a[i];
	}
	getMin();
	puts("");
	getMax();
	puts("");
	return 0;
}

最大子序和

给定一个长度为\(n\)的序列, 求出一段长度不超过\(m\)的连续子序列, 输出子序和.

不难想到前缀和操作, 在前缀和数组\(S\)中, 对于右端点\(r\), 需要找到左端点\(l\)满足如下条件: \[ \begin {cases} l \in [1, r - 1]&&&(1)\\ \min(S_l)&&&(2)\\ r - m \le l&&&(3) \end {cases} \] 其中, 3式的意思是让子序列的大小不超过\(m\).

#include <iostream>

const int N = 1e6;

using namespace std;

int m, n;
int a[N], s[N];

int q[N];

int main() {
	cin >> n >> m;
	for(int i = 1; i <= n; ++i) {
		cin >> a[i];
		s[i] = s[i - 1] + a[i];
	}

	int head = 0, tail = 0, ans = -0x3f;
	for(int i = 1; i <= n; ++i) {
	    while(head <= tail && i - m > q[head]) ++head;
	    ans = max(ans, s[i] - s[q[head]]);
		while(head <= tail && s[q[tail]] >= s[i]) --tail;
		q[++tail] = i;
	}
    
	cout << ans << endl;
	return 0;
}
